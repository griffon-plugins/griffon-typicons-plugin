364     private static final String ERROR_DESCRIPTION_BLANK = "Argument 'description' must not be blank";
365 
366     public static TypIconFont findByDescription(@Nonnull String description) {
367         requireNonBlank(description, "Icon description must not be blank.");
368 
369         String[] parts = description.split(":");
370         for (TypIconFont font : values()) {
371             if (font.getDescription().equals(parts[0])) {
372                 return font;
373             }
374         }
375 
376         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
377     }
378 
379     private String description;
380     private char code;
381 
382     TypIconFont(@Nonnull String description, char code) {
383         this.description = description;
384         this.code = code;
385     }
386 
387     @Nonnull
388     public String getDescription() {
389         return description;
390     }
391 
392     public char getCode() {
393         return code;
394     }

396     @Nonnull

398         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
399         throw new IllegalArgumentException("Description " + description + " is not a valid TypIcon icon description");
400     }
